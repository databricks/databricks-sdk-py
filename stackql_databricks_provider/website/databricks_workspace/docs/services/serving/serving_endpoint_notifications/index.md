```sql
UPDATE databricks_workspace.serving.serving_endpoint_notifications
SET 
email_notifications = '{{ email_notifications }}'
WHERE 
name = '{{ name }}' --required
AND workspace = '{{ workspace }}' --required
RETURNING
name,
email_notifications;
```
